15class MediatorMixin: 

16 """Mixin providing shared logic for both sync and async mediators. 

17 

18 This mixin contains the common initialization and request processing logic 

19 that is shared between the synchronous Mediator and asynchronous Mediator. 

20 

21 The actual send() method is implemented differently in each variant: 

22 - Synchronous: def send(...) -> ResponseT 

23 - Asynchronous: async def send(...) -> ResponseT 

24 

25 Attributes: 

26 _services: The services instance used to obtain handler and behavior instances. 

27 """ 

28 

29 _services: "ServiceProvider" 

30 

31 def __init__(self, services: "ServiceProvider") -> None: 

32 """Initialize mediator with services for obtaining handler and behavior instances. 

33 

34 Args: 

35 services: Any object implementing the ServiceProvider protocol. 

36 This can be a ServiceProvider from Services.provider(), 

37 a DependencyInjectorServiceProvider, or your own custom implementation. 

38 

39 Examples: 

40 ```python 

41 from pymediate import Mediator 

42 from pymediate.service import Services 

43 

44 services = Services() 

45 services.add(CreateUserHandler()) 

46 provider = services.provider() 

47 

48 mediator = Mediator(provider) 

49 ``` 

50 

51 With dependency injection: 

52 ```python 

53 from pymediate.providers import DependencyInjectorServiceProvider 

54 

55 container = AppContainer() 

56 provider = DependencyInjectorServiceProvider(container) 

57 mediator = Mediator(provider) 

58 ``` 

59 """ 

60 self._services = services 

61 

62 def _resolve_handler(self, request: "Request[Any]") -> Any: 

63 """Resolve the handler for a request. 

64 

65 Args: 

66 request: The request instance to process 

67 

68 Returns: 

69 Handler instance for the request 

70 

71 Raises: 

72 HandlerNotFoundError: If no handler is registered for the request type 

73 """ 

74 # Look up handler type from registry 

75 request_type = type(request) 

76 handler_class = registry.get_handler_class(request_type) 

77 if handler_class is None: 

78 raise errors.HandlerNotFoundError(request_type, []) 

79 

80 # Get handler instance 

81 return self._services.get(handler_class) 

82 

83 def _resolve_event_handlers(self, event: "Event") -> list[Any]: 

84 """Resolve every handler instance subscribed to an event. 

85 

86 Resolves all handler instances before any handler runs, so a resolution 

87 failure (a subscriber class with no registered instance) propagates 

88 immediately and never causes partial delivery. 

89 

90 Args: 

91 event: The event instance to publish 

92 

93 Returns: 

94 Handler instances in registration order, empty if none subscribed 

95 

96 Raises: 

97 ServiceNotFoundError: If a subscribed handler class has no 

98 registered instance in the service provider 

99 """ 

100 handler_classes = registry.get_event_handler_classes(type(event)) 

101 return [self._services.get(handler_class) for handler_class in handler_classes] 

102 

103 def _resolve_behaviors( 

104 self, request: "Request[Any]", pipeline_behavior_type: type 

105 ) -> list[Any]: 

106 """Resolve applicable behaviors for a request. 

107 

108 Args: 

109 request: The request instance to process 

110 pipeline_behavior_type: The PipelineBehavior type (sync or async variant) 

111 

112 Returns: 

113 List of applicable behavior instances 

114 """ 

115 # Get all registered pipeline behaviors 

116 all_behaviors: Sequence[Any] = self._services.get_all(pipeline_behavior_type) 

117 

118 # Filter behaviors to only those that apply to this request 

119 return [behavior for behavior in all_behaviors if type(behavior).should_apply(request)]
