9class EventHandler[EventT: Event](EventHandlerBaseMixin[EventT], ABC): 

10 """Abstract base class for asynchronous event handlers. 

11 

12 The async mirror of `pymediate.EventHandler`: any number of async event 

13 handlers may subscribe to the same event type, and the async 

14 `Mediator.publish()` runs all of them concurrently via `asyncio.gather` 

15 (tasks are created in registration order). 

16 

17 The handler performs class-definition-time validation via __init_subclass__ 

18 to ensure: 

19 - The __call__ method exists and is properly implemented 

20 - The __call__ method is asynchronous (async def) 

21 - The __call__ parameter annotates the exact declared event type 

22 (not a base class or union) 

23 - The __call__ return annotation is None - event handlers produce no response 

24 

25 This validation happens at class definition time (import time), catching 

26 errors early in the development cycle rather than at runtime. 

27 

28 Type Parameters: 

29 EventT: The type of event this handler subscribes to. Must inherit 

30 from Event; static type checkers enforce the bound, and PyMediate 

31 validates it at class definition time. 

32 

33 Examples: 

34 Two async handlers subscribed to one event: 

35 ```python 

36 import asyncio 

37 from dataclasses import dataclass 

38 from pymediate import Event, Services 

39 from pymediate.aio import EventHandler, Mediator 

40 

41 @dataclass 

42 class OrderPlaced(Event): 

43 order_id: int 

44 

45 class SendConfirmation(EventHandler[OrderPlaced]): 

46 async def __call__(self, event: OrderPlaced) -> None: 

47 print(f"confirming order {event.order_id}") 

48 

49 class UpdateAnalytics(EventHandler[OrderPlaced]): 

50 async def __call__(self, event: OrderPlaced) -> None: 

51 print(f"recording order {event.order_id}") 

52 

53 async def main(): 

54 services = Services() 

55 services.add(SendConfirmation()).add(UpdateAnalytics()) 

56 mediator = Mediator(services.provider()) 

57 

58 await mediator.publish(OrderPlaced(order_id=42)) 

59 

60 asyncio.run(main()) 

61 ``` 

62 

63 Note: 

64 Handlers for the same event run concurrently, so they must not rely on 

65 each other's effects or mutate shared state without synchronization. 

66 For synchronous event handlers, use `pymediate.EventHandler` instead. 

67 

68 Raises: 

69 InvalidHandlerSignatureError: If the __call__ signature is invalid or 

70 not async, including a return annotation other than None. 

71 InvalidEventTypeError: If the event type doesn't inherit from Event. 

72 

73 See Also: 

74 - Event: Base event class. 

75 - pymediate.aio.Mediator.publish: Publishes an event to all its handlers. 

76 - pymediate.EventHandler: Sync event handler variant. 

77 """ 

78

81 @abstractmethod 

82 async def __call__(self, event: EventT) -> None: 

83 """Handle the published event asynchronously. 

84 

85 This is an abstract method that must be implemented by all async 

86 EventHandler subclasses, with the signature 

87 `async def __call__(self, event: EventType) -> None: ...` 

88 

89 Args: 

90 event: The event to handle. 

91 

92 Note: 

93 The annotation must be the exact event class - a base class or 

94 union passes static checking (contravariance) but raises 

95 `InvalidHandlerSignatureError` at class definition. The return 

96 annotation must be `None`: publishing has no response, and any 

97 value a handler returns is discarded. This method must also be 

98 asynchronous (`async def`); for sync event handlers, use 

99 `pymediate.EventHandler`. 

100 """
